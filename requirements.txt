# ──────────────────────────────────────────────
# QwenRAG — Python dependencies

# Install with:  pip install -r requirements.txt

# Python 3.9 or newer is required.
# ──────────────────────────────────────────────

# ── Web UI ────────────────────────────────────
streamlit>=1.32.0

# ── PDF parsing ───────────────────────────────
pymupdf>=1.23.0          # PyMuPDF — fast, reliable PDF text extraction

# ── Embeddings ────────────────────────────────
sentence-transformers>=2.7.0   # Includes PyTorch CPU; downloads ~90 MB model on first run
                               # Model: all-MiniLM-L6-v2 (22M params, 384-dim)

# ── Vector search ─────────────────────────────
faiss-cpu>=1.7.4         # Pure CPU FAISS — no GPU needed
                         # For GPU machines (optional): faiss-gpu instead

# ── Hybrid retrieval ──────────────────────────
rank_bm25>=0.2.2         # BM25 keyword arm (fused with vectors via RRF)
                         # The cross-encoder reranker uses sentence-transformers
                         # (above) and downloads cross-encoder/ms-marco-MiniLM-L-6-v2
                         # (~80 MB) on first use. Disable via HYBRID_RETRIEVAL=false.

# ── HTTP client (for Ollama) ──────────────────
requests>=2.31.0

# ── Numerical computing ───────────────────────
numpy>=1.24.0

# ──────────────────────────────────────────────
# Packaged desktop app (the bundle) — NOT needed for dev
# ──────────────────────────────────────────────
# These power the self-contained, double-click build. In dev the app still uses
# Ollama + sentence-transformers (above); config auto-selects these only when
# frozen (LLM_BACKEND=llamacpp, EMBED_BACKEND=fastembed).
#
# llama-cpp-python        # In-process GGUF inference — replaces the Ollama daemon
# fastembed               # ONNX embeddings + reranker — replaces PyTorch (~2 GB smaller)
# pywebview               # Native window wrapping the local Streamlit server
platformdirs>=4.0.0       # Per-user writable data dir (uploads/index) when installed

pytest>=8.0.0            # Test suite:  python -m pytest tests/ -v
